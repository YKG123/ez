1. Two Pointers

Use When
  • Sorted arrays
  • Palindromes
  • Pair/triplet sums
Core Idea

Move pointers intelligently instead of nested loops.

Template
let l = 0; let r = arr.length - 1; while (l < r) { let sum = arr[l] + arr[r]; if (sum === target) { return [l, r]; } if (sum < target) l++; else r--; }
Example Problems
  • Two Sum II
  • Container With Most Water
  • Trapping Rain Water

3. Hash Map

Use When
  • Fast lookups
  • Counting frequencies
  • Duplicates/anagrams
Core Idea

Store previous information for O(1) access.

Template
const map = new Map(); for (let i = 0; i < nums.length; i++) { let needed = target - nums[i]; if (map.has(needed)) { return [map.get(needed), i]; } map.set(nums[i], i); }
Example Problems
  • Two Sum
  • Group Anagrams

4. Binary Search

Use When
  • Sorted arrays
  • Monotonic conditions
Core Idea

Cut search space in half each step.

Template
let l = 0; let r = arr.length - 1; while (l <= r) { let mid = Math.floor((l + r) / 2); if (arr[mid] === target) return mid; if (arr[mid] < target) l = mid + 1; else r = mid - 1; }
Example Problems
  • Search Rotated Array
  • Koko Eating Bananas

5. DFS

Use When
  • Trees
  • Graphs
  • Backtracking
Core Idea

Go deep before backtracking.

Template
function dfs(node) { if (!node) return; dfs(node.left); dfs(node.right); }
Example Problems
  • Max Depth Binary Tree
  • Number of Islands

6. BFS

Use When
  • Shortest path
  • Level-order traversal
Core Idea

Process nodes level by level.

Template
let queue = [start]; while (queue.length) { let node = queue.shift(); for (let neighbor of node.neighbors) { queue.push(neighbor); } }
Example Problems
  • Word Ladder
  • Rotting Oranges

9. Monotonic Stack

Use When
  • Next greater/smaller element
  • Histogram problems
Core Idea

Maintain increasing/decreasing order in stack.

Template
const stack = []; for (let i = 0; i < arr.length; i++) { while ( stack.length && arr[i] > arr[stack[stack.length - 1]] ) { stack.pop(); } stack.push(i); }
Example Problems
  • Daily Temperatures

11. Prefix Sum

Use When
  • Range sum queries
  • Subarray sums
  • Cumulative computations
Core Idea

Store cumulative totals to answer ranges quickly.

Template
const prefix = Array(nums.length + 1).fill(0); for (let i = 0; i < nums.length; i++) { prefix[i + 1] = prefix[i] + nums[i]; } // Sum from L to R let sum = prefix[R + 1] - prefix[L];

12. Backtracking

Use When
  • Generate all possibilities
  • Permutations/combinations
  • Constraint search
Core Idea

Choose → recurse → undo.

Template
function backtrack(path) { if (done(path)) { result.push([...path]); return; } for (let choice of choices) { path.push(choice); backtrack(path); path.pop(); } }
Example Problems
  • Subsets
  • Permutations
  • N-Queens

13. Union Find

Use When
  • Connected components
  • Graph connectivity
  • Cycle detection
Core Idea

Efficiently merge and track groups.

Template
function find(x) { if (parent[x] !== x) { parent[x] = find(parent[x]); } return parent[x]; } function union(a, b) { parent[find(a)] = find(b); }
Example Problems
  • Accounts Merge
  • Connected Components

14. Fast & Slow Pointers

Use When
  • Linked lists
  • Cycle detection
  • Middle node problems
Core Idea

Move pointers at different speeds.

Template
let slow = head; let fast = head; while (fast && fast.next) { slow = slow.next; fast = fast.next.next; if (slow === fast) { return true; } }
Example Problems
  • Linked List Cycle
